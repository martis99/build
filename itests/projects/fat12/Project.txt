NAME = fat12
TYPE = FAT12
DEPENDS = bin, elf
HEADER = bin
FILES = elf
FILENAME = floppy
REQUIRE = qemu-system-i386, qemu-system-x86
RUN = qemu-system-$(ARCH) -nographic -drive format=raw,file=$(TARGET_FAT12) -device isa-debug-exit,iobase=0xf4,iosize=0x4 || [ $$? -eq 1 ]
DRUN = qemu-system-$(ARCH) -nographic -drive format=raw,file=$(TARGET_FAT12) -device isa-debug-exit,iobase=0xf4,iosize=0x4 || [ $$? -eq 1 ]
ARTIFACT = floppy.img
